public class Digit {
    private ArrayList<Integer> digitList;

    public Digit(int num) {
        digitList = new ArrayList<Integer>();
        int temp = num, nod = 0;
        while(temp > 0) {
            digitList.add(temp % 10);
            temp = temp/10;
            nod++;
        }
        int[] arr = new int[nod];
        for(int i=0; i < arr.length; i++) {
            arr[i] = digitList.get(nod-1);
            nod--;
        }
        digitList.clear();
        for(int i = 0; i < arr.length; i++) {
            digitList.add(i, arr[i]);
        }
    }

    public boolean isStrictlyIncreasing() {
        for(int i = 1; i < digitList.size(); i++) {
            // if(digitList.get(i) > digitList.get(i-1)) {}
            // else{
            //     return false;
            // }
            if(!(digitList.get(i) > digitList.get(i-1)))
                return false;
        }
        return true;
    }
    
    public void print() {
        for(int i : digitList) {
            System.out.println(i);
        }
    }

    public static void main(String[] args) {
        Digit d1 = new Digit(0);
        
        d1.print();
        System.out.println(d1.isStrictlyIncreasing());
    }
}

public class ArrayMethods {
    private int[] values;

    public ArrayMethods(int[] values) {
        this.values = values;
    }

    public void print() {
        for(int i = 0; i < values.length; i++) {
            System.out.println(values[i]);
        }
    } 
    public void swap() {
        int x = values[0];
        values[0]= values[values.length-1];
        values[values.length-1] = x;

        print();
    }

    public static void main(String[] args) {
        ArrayMethods r = new ArrayMethods(new int[] {20,30,40,60});
        System.out.println(r);
        r.swap();
    }
}
