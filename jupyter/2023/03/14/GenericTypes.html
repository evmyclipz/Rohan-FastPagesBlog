    Implementation of the LinkedList in Stack</h3>

    In this implementation, Stack class uses the LinkedList interface to provide basic stack operations such as push, pop, peek, isEmpty, and size. The LinkedList interface methods add, remove, get, and indexOf are not implemented, as they are not needed for this implementation of a stack.</p> </div> </div> </div> 

    public class Stack<T> implements LinkedList<T> {

    private LinkedList<T> top;

    // Constructor
    public Stack() {
        this.top = null;
    }

    // Add a new item to the top of the stack
    public void push(T item) {
        LinkedList<T> node = new LinkedList<>(item, null);
        if (this.top != null) {
            node.setNextNode(this.top);
            this.top.setPrevNode(node);
        }
        this.top = node;
    }

    // Remove and return the top item from the stack
    public T pop() {
        if (this.top == null) {
            throw new NoSuchElementException();
        }
        T item = this.top.getData();
        this.top = this.top.getNext();
        if (this.top != null) {
            this.top.setPrevNode(null);
        }
        return item;
    }

    // Return the top item from the stack without removing it
    public T peek() {
        if (this.top == null) {
            throw new NoSuchElementException();
        }
        return this.top.getData();
    }

    // Check if the stack is empty
    public boolean isEmpty() {
        return this.top == null;
    }

    // Get the size of the stack
    public int size() {
        int size = 0;
        LinkedList<T> current = this.top;
        while (current != null) {
            size++;
            current = current.getNext();
        }
        return size;
    }

    // Implement the LinkedList<T> interface method
    public void add(T item) {
        this.push(item);
    }

}
