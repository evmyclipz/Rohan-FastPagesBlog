public class ArrayResizer {

    /** Returns true if and only if every value in row r of array2D
    * is non-zero.
    * Precondition: r is a valid row index in array2D.
    * Postcondition: array2D is unchanged.
    */
    public static boolean isNonZeroRow(int[][] arr, int r) { 
        /* to be implemented in part (a) */ 
        for(int i = 0; i < arr[r].length; i++) {
            if(arr[r][i] == 0) {
                return false;
            }
        }
        return true;
    }

    
    /** Returns the number of rows in array2D that contain all
    * non-zero values.
    * Postcondition: array2D is unchanged.
    */
    public static int numNonZeroRows(int[][] arr) { 
        /* implementation not shown */
        int count = 0;
        for(int i = 0; i < arr.length; i++) {
            if(isNonZeroRow(arr, i)) {
                count++;
            }
        }
        return count; 
    }

    

    /** Returns a new, possibly smaller, two-dimensional array that
    * contains only rows from array2D with no zeros, as described
    * in part (b).
    * Precondition: array2D contains at least one column
    * and at least one row with no zeros.
    * Postcondition: array2D is unchanged
    */
    public static int[][] resize(int[][] arr) { 
        /* to be implemented in part (b) */
        int row = numNonZeroRows(arr);
        //System.out.println("Row: "+row);
        int col = arr[0].length;
        //System.out.println("Col: "+col);
        int r = 0;
        int c = 0;
        int[][] resizedArr = new int[row][col]; 
        for(int i = 0; i < arr.length;i++) {

            for(int j = 0; j < arr[i].length;j++) {

                if(isNonZeroRow(arr, i)) {
                    //System.out.println(arr[i][j]);
                    resizedArr[r][c++] = arr[i][j];

                }
            }
            if(isNonZeroRow(arr, i)) {
                r++;
                c = 0;
            }
        }
        return resizedArr;
    }

    public static void main(String[] args) {
        int[][] array = {{2, 1, 0},
                        {1, 3, 2},
                        {0, 0, 0},
                        {4, 5, 6}};

        //System.out.print(isNonZeroRow(array, 3));

        int[][] newArr = resize(array);

        for (int i = 0; i < newArr.length; i++) {
            for (int j = 0; j < newArr[i].length; j++) {
                System.out.print(newArr[i][j] + " ");
            }
            System.out.println();
        }
        
        /* 
        for(int i : test) {
            System.out.println(i);
        }
        */
    }

}

import java.util.Random;
import java.util.*;

public class ClubMembers {

    private ArrayList<MemberInfo> memberList;

    
    public ClubMembers() {
        this.memberList = new ArrayList<MemberInfo>();
    }

    /** Adds new club members to memberList, as described in part (a).
    * Precondition: names is a non-empty array.
    */

    public void addMembers(String[] names, int gradYear) { 
        /* to be implemented in part (a) */ 
        Random rd = new Random();
        for(String name : names) {
            memberList.add(new MemberInfo(name, gradYear, rd.nextBoolean()));
        }
    } 

    /** Removes members who have graduated and returns a list of
    * members who have graduated and are in good standing,
    * as described in part (b).
    */

    public ArrayList<MemberInfo> removeMembers(int year) { 
        /* to be implemented in part (b) */ 
        ArrayList<MemberInfo> graduated = new ArrayList<MemberInfo>();
        ArrayList<MemberInfo> goodStanding = new ArrayList<MemberInfo>();
        for(MemberInfo member : memberList) {
            if(member.getGradYear() <= year)  {
                graduated.add(member);
                if(member.inGoodStanding() == true) {
                    goodStanding.add(member);
                }
            }   
        }
         
        for(MemberInfo member : graduated) {
            memberList.remove(member);
        }
        
        return goodStanding;
    }

    public void print() {
        for(MemberInfo member : memberList) {
            System.out.println("Member: "+member.getName()+" Grad year: "+member.getGradYear()+" Good Standing: "+member.inGoodStanding());
        }
    }

    public static void main(String[] args) {
        ClubMembers club = new ClubMembers();
        String[] names = {"Rohan", "Rohit", "Arnav", "Mohit", "Rahul", "Jason"};
        club.addMembers(names, 2023);
        String[] names1 = {"Bahal", "Yash", "Daniel", "Aadit", "Mom", "King"};
        club.addMembers(names1, 2021);
        
        System.out.println("Before removal of 2023 batch");
        club.print();
        System.out.println();
        System.out.println("Graduated class with good standing");
        ArrayList<MemberInfo> graduated = club.removeMembers(2021);
        for(MemberInfo member : graduated) {
            System.out.println("Member: "+member.getName()+" Grad year: "+member.getGradYear()+" Good Standing: "+member.inGoodStanding());
        }
        System.out.println();
        System.out.println("After removal of 2023 batch");
        club.print();
    }

}
